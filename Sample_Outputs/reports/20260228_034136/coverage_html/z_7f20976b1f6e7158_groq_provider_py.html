85 async def generate_completion( 

86 self, 

87 messages: list[AIMessage], 

88 model: str = "llama-3.1-70b-versatile", 

89 temperature: float = 0.7, 

90 max_tokens: int = 2000, 

91 **kwargs 

92 ) -> AIResponse: 

93 """ 

94 Generate completion using Groq API. 

95 

96 Groq uses OpenAI-compatible API, so message format is identical. 

97 """ 

98 try: 

99 # Convert AIMessage to Groq format (same as OpenAI) 

100 groq_messages = [ 

101 {"role": msg.role, "content": msg.content} 

102 for msg in messages 

103 ] 

104 

105 logger.info( 

106 f"Groq request: model={model}, messages={len(groq_messages)}, " 

107 f"max_tokens={max_tokens}" 

108 ) 

109 

110 response = await self.client.chat.completions.create( 

111 model=model, 

112 messages=groq_messages, 

113 temperature=temperature, 

114 max_tokens=max_tokens, 

115 **kwargs 

116 ) 

117 

118 content = response.choices[0].message.content 

119 tokens_used = response.usage.total_tokens 

120 

121 logger.info( 

122 f"Groq response: tokens={tokens_used}, speed=ULTRA-FAST ⚡, " 

123 f"cost=FREE 🎉, finish_reason={response.choices[0].finish_reason}" 

124 ) 

125 

126 return AIResponse( 

127 content=content, 

128 model=model, 

129 tokens_used=tokens_used, 

130 provider="groq", 

131 finish_reason=response.choices[0].finish_reason, 

132 cost_usd=0.0, # FREE tier! 

133 ) 

134 

135 except RateLimitError as e: 

136 logger.error(f"Groq rate limit: {e}") 

137 raise AIRateLimitError(f"Groq rate limit exceeded: {e}") 

138 

139 except AuthenticationError as e: 

140 logger.error(f"Groq auth error: {e}") 

141 raise AIAuthenticationError(f"Invalid Groq API key: {e}") 

142 

143 except Exception as e: 

144 logger.error(f"Groq error: {e}") 

145 raise AIProviderError(f"Groq API error: {e}")
