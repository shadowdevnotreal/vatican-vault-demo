24# WebSocket connection manager 

25class ConnectionManager: 

26 """Manage WebSocket connections for real-time updates.""" 

27 

28 def __init__(self): 

29 self.active_connections: list[WebSocket] = [] 

30 

31 async def connect(self, websocket: WebSocket): 

32 """Accept new WebSocket connection.""" 

33 await websocket.accept() 

34 self.active_connections.append(websocket) 

35 logger.info(f"WebSocket client connected. Total: {len(self.active_connections)}") 

36 

37 def disconnect(self, websocket: WebSocket): 

38 """Remove WebSocket connection.""" 

39 self.active_connections.remove(websocket) 

40 logger.info(f"WebSocket client disconnected. Total: {len(self.active_connections)}") 

41 

42 async def broadcast(self, message: dict): 

43 """Broadcast message to all connected clients.""" 

44 for connection in self.active_connections: 

45 try: 

46 await connection.send_json(message) 

47 except Exception as e: 

48 logger.error(f"Error broadcasting to client: {e}")

134 # WebSocket endpoint 

135 @app.websocket("/ws") 

136 async def websocket_endpoint(websocket: WebSocket): 

137 """ 

138 WebSocket endpoint for real-time activity monitoring. 

139 

140 Clients can subscribe to activity streams and receive updates in real-time. 

141 """ 

142 await manager.connect(websocket) 

143 

144 try: 

145 while True: 

146 # Receive messages from client 

147 data = await websocket.receive_json() 

148 

149 # Handle different message types 

150 if data.get("type") == "subscribe": 

151 # Subscribe to activity feed 

152 await websocket.send_json({ 

153 "type": "subscribed", 

154 "message": "Successfully subscribed to activity feed" 

155 }) 

156 

157 elif data.get("type") == "ping": 

158 # Respond to ping 

159 await websocket.send_json({ 

160 "type": "pong", 

161 "timestamp": data.get("timestamp") 

162 }) 

163 

164 except WebSocketDisconnect: 

165 manager.disconnect(websocket) 

166 except Exception as e: 

167 logger.error(f"WebSocket error: {e}") 

168 manager.disconnect(websocket) 

169 

170 return app
