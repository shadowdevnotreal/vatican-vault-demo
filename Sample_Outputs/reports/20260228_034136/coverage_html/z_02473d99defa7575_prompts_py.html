17def build_activity_timeline( 

18 activities: list[UnifiedActivity], 

19 max_activities: int = 100, 

20 include_ocr: bool = True 

21) -> str: 

22 """ 

23 Build formatted timeline of activities for AI prompt. 

24 

25 Args: 

26 activities: List of unified activities 

27 max_activities: Maximum activities to include (prevent token overflow) 

28 include_ocr: Whether to include OCR text excerpts 

29 

30 Returns: 

31 Formatted timeline string 

32 """ 

33 timeline_lines = [] 

34 

35 # Limit activities if too many 

36 activities_to_process = activities[:max_activities] if len(activities) > max_activities else activities 

37 

38 for activity in activities_to_process: 

39 timestamp = activity.timestamp.strftime("%H:%M:%S") 

40 app_name = activity.app_name or "Unknown" 

41 window_title = activity.content or activity.window_title or "No title" 

42 

43 # Build activity line 

44 line = f"{timestamp} | {app_name} | {window_title}" 

45 

46 # Add OCR excerpt if available 

47 if include_ocr and activity.ocr_text: 

48 # Truncate OCR text to first 100 chars 

49 ocr_excerpt = activity.ocr_text[:100] 

50 if len(activity.ocr_text) > 100: 

51 ocr_excerpt += "..." 

52 line += f" | OCR: {ocr_excerpt}" 

53 

54 timeline_lines.append(line) 

55 

56 if len(activities) > max_activities: 

57 timeline_lines.append(f"\n... [{len(activities) - max_activities} more activities omitted] ...") 

58 

59 return "\n".join(timeline_lines)

62def build_security_analysis_prompt( 

63 activities: list[UnifiedActivity], 

64 detail_level: str = "detailed", 

65 sensitive_data: Optional[list[dict]] = None, 

66) -> list[AIMessage]: 

67 """ 

68 Build prompt for security-focused session analysis. 

69 

70 Instructs AI to find threats, breaches, suspicious activities. 

71 """ 

72 # Build activity timeline 

73 timeline = build_activity_timeline(activities, max_activities=100) 

74 

75 # Calculate session metadata 

76 start_time = activities[0].timestamp if activities else datetime.now() 

77 end_time = activities[-1].timestamp if activities else datetime.now() 

78 duration_hours = (end_time - start_time).total_seconds() / 3600 

79 

80 # Build sensitive data summary 

81 sensitive_summary = "" 

82 if sensitive_data: 

83 sensitive_summary = f"\n\nDETECTED SENSITIVE DATA:\n" 

84 for item in sensitive_data[:20]: # Limit to 20 

85 sensitive_summary += f"- {item.get('type', 'unknown')}: {item.get('redacted_value', 'N/A')} (severity: {item.get('severity', 'unknown')})\n" 

86 

87 system_message = AIMessage( 

88 role="system", 

89 content="""You are an expert cybersecurity analyst conducting a forensic investigation. 

90 

91Analyze Windows Recall session data for: 

92- Security threats and vulnerabilities 

93- Data breaches and credential exposure 

94- Insider threat indicators 

95- Policy violations 

96- Suspicious patterns 

97 

98Be specific with timestamps and evidence. Classify severity as critical/high/medium/low.""" 

99 ) 

100 

101 user_message = AIMessage( 

102 role="user", 

103 content=f"""Analyze this user session for security threats: 

104 

105SESSION METADATA: 

106- Time Range: {start_time.strftime('%Y-%m-%d %H:%M')} to {end_time.strftime('%Y-%m-%d %H:%M')} 

107- Duration: {duration_hours:.1f} hours 

108- Activities Captured: {len(activities)} 

109{sensitive_summary} 

110 

111ACTIVITY TIMELINE: 

112{timeline} 

113 

114ANALYSIS REQUIREMENTS: 

1151. EXECUTIVE SUMMARY: 2-3 sentences summarizing key security concerns 

1162. SECURITY ALERTS: List specific threats with timestamps and severity 

1173. RISK ASSESSMENT: Overall risk score (0.0-1.0) with justification 

1184. RECOMMENDATIONS: Prioritized actionable steps 

119 

120Format your response with clear sections using markdown headers.""" 

121 ) 

122 

123 return [system_message, user_message]
